[https]
client = no
cert = /etc/stunnel/stunnel.pem
key = /etc/stunnel/stunnel.pem
accept=0.0.0.0:4128      #等待远端访问端口
connect=localhost:3128   #连接到本地squid端口
---


#CLIENT
[https]
client = yes
accept = 127.0.0.1:8888     #本地8888端口等待访问
connnect=server_IP:4128     #连接服务端4128端口
